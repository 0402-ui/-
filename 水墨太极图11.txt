        // 模式控制变量
        let currentMode = 1;       // 1: 太极模式, 2: 阴阳分流模式
        let modeLerp = 0;          // 模式形态切换插值 (0~1)
        
        // 黑白反转控制 (流水模式专用)
        let targetColorSwap = 0;   // 目标颜色反转状态 (0: 正常, 1: 反转)
        let colorSwapLerp = 0;     // 颜色平滑渐变系数

        let mouseEnergy = 0;

        let dragonLerp = 0;

        };

        // 自定义颜色基础变量

        let c1BaseR = 20, c1BaseG = 20, c1BaseB = 20;     // 阴鱼(默认黑)基色
        let c2BaseR = 240, c2BaseG = 240, c2BaseB = 240;  // 阳鱼(默认白)基色

        // 发光特效控制变量
        let isGlowMode = false;

        // 3D 模式控制变量
        let is3DMode = false;
        let lerp3D = 0;
        let rotX = 0;
        let rotY = 0;
        let targetRotX = 0;
        let targetRotY = 0;
        let currentCameraZ = 200; // 默认将镜头大幅拉近，增加原始冲击力（原本是800）
        let targetCameraZ = 200;
        let initialPinchDist = 0; // 用于移动端双指缩放

        // 十六进制颜色转换RGB辅助函数
        function hexToRgb(hex) {
            let result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
            return result ? {
                r: parseInt(result[1], 16),
                g: parseInt(result[2], 16),
                b: parseInt(result[3], 16)
            } : null;
        }

        setTimeout(() => {
            if (hintText) hintText.style.opacity = '0';
        }, 5000);

        function resize() {
            width = canvas.width = window.innerWidth;
            height = canvas.height = window.innerHeight;
        }

        function getTaiChiColor(x, y) {
            let d = Math.hypot(x, y);
            if (d > R) return null; 

            if (Math.hypot(x, y + R / 2) < R / 6) return 'black';
            if (Math.hypot(x, y - R / 2) < R / 6) return 'white';
            if (Math.hypot(x, y + R / 2) < R / 2) return 'white';
            if (Math.hypot(x, y - R / 2) < R / 2) return 'black';

            return x > 0 ? 'black' : 'white';
        }

        class Particle {
            constructor(baseX, baseY, colorType) {
                this.colorType = colorType; 
                this.isLeft = (colorType === 'black'); // 默认黑色在左，白色在右

                this.baseX = baseX + (Math.random() - 0.5) * 6;
                this.baseY = baseY + (Math.random() - 0.5) * 6;
                
                this.x = width / 2 + (Math.random() - 0.5) * width;
                this.y = height / 2 + (Math.random() - 0.5) * height;
                
                this.vx = 0;
                this.vy = 0;
                this.size = Math.random() * 2 + 0.6; 
                
                this.baseSpring = Math.random() * 0.015 + 0.01; 
                this.friction = Math.random() * 0.05 + 0.86; 

                // --- 模式二：垂直河流流场初始参数 ---
                this.m2_vx = 0;
                this.m2_vy = 0;
                this.m2_speed = Math.random() * 2.5 + 1.5; 
                this.m2_offsetX_ratio = Math.random(); // 横向分布比例 (0~1)
                
                // 3D 渲染所需属性
                this.zSign = Math.random() > 0.5 ? 1 : -1; // 随机分布在球体的前半球或后半球
                this.projX = 0;
                this.projY = 0;
                this.projZ = 0;
                this.projScale = 1;

                // 初始位置分布 (铺满全屏)
                if (this.isLeft) {
                    this.m2_x = (window.innerWidth / 2) * this.m2_offsetX_ratio;
                } else {
                    this.m2_x = window.innerWidth / 2 + (window.innerWidth / 2) * this.m2_offsetX_ratio;
                }
                this.m2_y = Math.random() * window.innerHeight;
                // ------------------------------------

                let dist = Math.hypot(baseX, baseY);
                this.ratio = Math.min(dist / R, 1);

                // 保存基础RGB色彩值，加入微小随机噪点产生水墨颗粒感
                if (colorType === 'black') {
                    this.base_r = Math.max(0, Math.min(255, c1BaseR + (Math.random() * 30 - 15)));
                    this.base_g = Math.max(0, Math.min(255, c1BaseG + (Math.random() * 30 - 15)));
                    this.base_b = Math.max(0, Math.min(255, c1BaseB + (Math.random() * 30 - 15)));
                    this.a = Math.random() * 0.5 + 0.2;
                } else {
                    this.base_r = Math.max(0, Math.min(255, c2BaseR + (Math.random() * 20 - 10)));
                    this.base_g = Math.max(0, Math.min(255, c2BaseG + (Math.random() * 20 - 10)));
                    this.base_b = Math.max(0, Math.min(255, c2BaseB + (Math.random() * 20 - 10)));
                    this.a = Math.random() * 0.5 + 0.25;
                }
            }

            update(cx, cy, currentAngle) {
                // 1. 计算模式一 (太极) 的目标坐标
                let totalAngle = currentAngle + angleOffset;
                let taiChiX = cx + this.baseX * Math.cos(totalAngle) - this.baseY * Math.sin(totalAngle);
                let taiChiY = cy + this.baseX * Math.sin(totalAngle) + this.baseY * Math.cos(totalAngle);
                
                // 2. 计算模式二 (双流) 的运动轨迹 (基于手绘图流体动力学)
                // 动态的波浪中轴线
                let divideX = width / 2 + Math.sin(this.m2_y * 0.005 + currentAngle * 1.5) * (width * 0.12);
                
                if (this.isLeft) {
                    // 左侧：往下流 (y增加)
                    this.m2_vy = this.m2_speed;
                    // 流场引力：根据比例铺满 0 到 divideX 之间的空间，留出15px中缝
                    let targetX = (divideX - 15) * this.m2_offsetX_ratio;
                    let targetVx = (targetX - this.m2_x) * 0.015;
                    this.m2_vx += (targetVx - this.m2_vx) * 0.05; 
                    
                    // 触底后从左上角重新下落
                    if (this.m2_y > height + 50) {
                        this.m2_y = -50;
                        this.m2_offsetX_ratio = Math.random(); // 重新随机分布比例
                        this.m2_x = (width / 2) * this.m2_offsetX_ratio;
                    }
                } else {
                    // 右侧：往上流 (y减小)
                    this.m2_vy = -this.m2_speed;
                    // 流场引力：根据比例铺满 divideX 到 width 之间的空间，留出15px中缝
                    let targetX = (divideX + 15) + (width - divideX - 15) * this.m2_offsetX_ratio;
                    let targetVx = (targetX - this.m2_x) * 0.015;
                    this.m2_vx += (targetVx - this.m2_vx) * 0.05;
                    
                    // 触顶后从右下角重新升起
                    if (this.m2_y < -50) {
                        this.m2_y = height + 50;
                        this.m2_offsetX_ratio = Math.random(); // 重新随机分布比例
                        this.m2_x = width / 2 + (width / 2) * this.m2_offsetX_ratio;
                    }
                }

                this.m2_x += this.m2_vx;
                this.m2_y += this.m2_vy;

                // 3. 模式融合与切换的平滑插值 (太极 <-> 双流)
                let baseTargetX = taiChiX * (1 - modeLerp) + this.m2_x * modeLerp;
                let baseTargetY = taiChiY * (1 - modeLerp) + this.m2_y * modeLerp;

                let targetX = baseTargetX;
                let targetY = baseTargetY;

                // 墨龙形态 (兼容双模式)
                if (dragonLerp > 0.01 && dragonSpine.length > 5) {
                    let spineIdx = Math.floor(this.ratio * (dragonSpine.length - 1));
                    let pt = dragonSpine[spineIdx];
                    if (pt) {
                        let scale = (1 - this.ratio) * 0.5 + 0.1; 
                        let dragonX = pt.x + this.baseX * scale;
                        let dragonY = pt.y + this.baseY * scale;
                        
                        targetX = baseTargetX * (1 - dragonLerp) + dragonX * dragonLerp;
                        targetY = baseTargetY * (1 - dragonLerp) + dragonY * dragonLerp;
                    }
                }

                // 研墨融合模式物理影响
                let currentSpring = this.baseSpring * (1 - mixLerp * 0.85); 
                
                if (mixLerp > 0.01) {
                    let dxCenter = this.x - cx;
                    let dyCenter = this.y - cy;
                    let distCenter = Math.hypot(dxCenter, dyCenter) || 1;
                    
                    this.vx += (-dyCenter / distCenter) * mixLerp * 1.5;
                    this.vy += (dxCenter / distCenter) * mixLerp * 1.5;
                    this.vx -= (dxCenter / distCenter) * mixLerp * 0.2;
                    this.vy -= (dyCenter / distCenter) * mixLerp * 0.2;
                }

                // 弹簧力
                let dx = targetX - this.x;
                let dy = targetY - this.y;
                this.vx += dx * currentSpring;
                this.vy += dy * currentSpring;

                // 鼠标排斥/搅动力 (仅在2D模式下生效，防止3D模式下破坏星球结构的稳定)
                if (mouse.isActive && !is3DMode) {
                    let mdx = this.x - mouse.x;
                    let mdy = this.y - mouse.y;
                    let dist = Math.hypot(mdx, mdy);

                    if (dist < mouse.radius) {
                        let force = (mouse.radius - dist) / mouse.radius;
                        let swirlX = -mdy / dist;
                        let swirlY = mdx / dist;
                        let pushX = mdx / dist;
                        let pushY = mdy / dist;

                        this.vx += (pushX * 1.8 + swirlX * 3.2) * force;
                        this.vy += (pushY * 1.8 + swirlY * 3.2) * force;
                    }
                }

                // 布朗运动与阻力
                this.vx += (Math.random() - 0.5) * 0.2;
                this.vy += (Math.random() - 0.5) * 0.2;

                this.vx *= this.friction;
                this.vy *= this.friction;
                this.x += this.vx;
                this.y += this.vy;

                // --- 3D 投影逻辑 (伪 3D 引擎) ---
                let dx3d = this.x - cx;
                let dy3d = this.y - cy;
                let dist2d = Math.hypot(dx3d, dy3d);
                
                // 将 2D 圆面平滑映射到 3D 球体表面
                let mappedZ = 0;
                if (dist2d < R * 1.5) {
                    let ratio = Math.min(1, dist2d / R);
                    mappedZ = Math.sqrt(1 - ratio * ratio) * R * this.zSign;
                }
                
                let currentZ = mappedZ * lerp3D;
                
                // 3D 矩阵旋转 (绕 X 轴和 Y 轴)
                let cosX = Math.cos(rotX), sinX = Math.sin(rotX);
                let cosY = Math.cos(rotY), sinY = Math.sin(rotY);
                
                let y1 = dy3d * cosX - currentZ * sinX;
                let z1 = dy3d * sinX + currentZ * cosX;
                
                let x2 = dx3d * cosY + z1 * sinY;
                let z2 = -dx3d * sinY + z1 * cosY;
                
                // 透视投影计算与深度缩放
                let focalLength = 500;
                // 防止摄像机穿模(拉得太近)时除以负数导致画面撕裂，限制最小分母距离为10
                let distance = Math.max(10, focalLength + currentCameraZ + z2);
                let perspectiveScale = focalLength / distance;
                
                // 在 2D 和 3D 状态间平滑插值过渡
                this.projScale = 1 * (1 - lerp3D) + perspectiveScale * lerp3D;
                this.projX = cx + x2 * this.projScale;
                this.projY = cy + y1 * this.projScale;
                this.projZ = z2 * lerp3D;
            }

            draw() {
                // --- 颜色处理逻辑 ---
                let tr = this.base_r;
                let tg = this.base_g;
                let tb = this.base_b;

                // 处理模式二的平滑黑白互换
                if (colorSwapLerp > 0) {
                    // 黑转白，白转黑 (约等于255 - 基础色)
                    let inv_r = 255 - this.base_r;
                    let inv_g = 255 - this.base_g;
                    let inv_b = 255 - this.base_b;

                    tr = this.base_r * (1 - colorSwapLerp) + inv_r * colorSwapLerp;
                    tg = this.base_g * (1 - colorSwapLerp) + inv_g * colorSwapLerp;
                    tb = this.base_b * (1 - colorSwapLerp) + inv_b * colorSwapLerp;
                }
                
                // 处理搅动时的无极灰融合
                let grayVal = 135;
                let final_r = Math.floor(tr * (1 - mixLerp) + grayVal * mixLerp);
                let final_g = Math.floor(tg * (1 - mixLerp) + grayVal * mixLerp);
                let final_b = Math.floor(tb * (1 - mixLerp) + grayVal * mixLerp);
                
                // 灵光模式下，进一步降低粒子的透明度（从0.7降至0.35），防止多粒子叠加时过度曝光（太亮）
                let final_a = isGlowMode ? this.a * 0.35 : this.a;

                // 3D 模式下：根据 Z 轴深度产生大气透视感（远处的更暗更透明）
                if (lerp3D > 0.01) {
                    let depthFactor = 1 - (this.projZ / (R * 2)); 
                    final_a *= Math.max(0.1, Math.min(1.2, depthFactor));
                }

                // 稍微减小发光模式下的尺寸膨胀（从1.5降至1.3），让光晕更收敛
                // 并且乘以 3D 透视缩放比例
                let final_size = (isGlowMode ? this.size * 1.3 : this.size) * this.projScale;

                ctx.fillStyle = `rgba(${final_r}, ${final_g}, ${final_b}, ${final_a})`;

                ctx.beginPath();
                ctx.arc(this.projX, this.projY, Math.max(0.1, final_size), 0, Math.PI * 2);
                ctx.fill();
            }
        }

        function initParticles() {
            particles = [];
            const numParticles = 6500; 
            let attempts = 0;

            while (particles.length < numParticles && attempts < 50000) {
                attempts++;
                let x = (Math.random() * 2 - 1) * R;
                let y = (Math.random() * 2 - 1) * R;
                
                let colorType = getTaiChiColor(x, y);
                if (colorType !== null) {
                    particles.push(new Particle(x, y, colorType));
                }
            }
        }

        function animate() {
            // 背景颜色平滑过渡到灰色
            let bgR = Math.floor(bgBaseR * (1 - mixLerp) + 135 * mixLerp);
            let bgG = Math.floor(bgBaseG * (1 - mixLerp) + 135 * mixLerp);
            let bgB = Math.floor(bgBaseB * (1 - mixLerp) + 135 * mixLerp);
            
            // 灵光模式下，稍微提高背景刷新的不透明度（从0.08调高到0.12），减少光晕的过度堆积
            let trailAlpha = isGlowMode ? 0.12 : 0.2;

            document.body.style.backgroundColor = `rgb(${bgR}, ${bgG}, ${bgB})`;
            ctx.fillStyle = `rgba(${bgR}, ${bgG}, ${bgB}, ${trailAlpha})`; 
            ctx.fillRect(0, 0, width, height);

            // 更新模式切换插值
            let targetModeLerp = (currentMode === 1) ? 0 : 1;
            modeLerp += (targetModeLerp - modeLerp) * 0.03;

            // 更新颜色互换插值 (流水模式专属特效)
            colorSwapLerp += (targetColorSwap - colorSwapLerp) * 0.04;

            angleOffset += (targetAngleOffset - angleOffset) * 0.06; 
            
            // 状态与能量控制
            mouseEnergy *= 0.94; 
            if (mouseEnergy > 120 && !isDragging) isDragonMode = true;
            if (mouseEnergy < 20 || isDragging) isDragonMode = false;

            // 触发自动恢复的无极(灰)彩蛋模式
            if (!isZenMode && mixEnergy > 480) {
                isZenMode = true;
                ignoreDrag = true; 
                setTimeout(() => {
                    isZenMode = false;
                    mixEnergy = 0; 
                }, 8000); // 修改这里：将 4000 改为 8000，使彩蛋文案持续8秒
            }

            if (isZenMode) {
                mixEnergy = 500; 
            } else if (!isDragging || ignoreDrag) {
                mixEnergy *= 0.96; 
            }
            
            // 达到350能量即进入完全融合的混沌状态
            let targetMix = mixEnergy > 350 ? 1 : (mixEnergy / 350); 
            targetMix = Math.min(1, targetMix);
            mixLerp += (targetMix - mixLerp) * 0.02; 

            // 根据当前模式控制不同彩蛋文字的显示
            if (mixLerp > 0.8) {
                if (currentMode === 1) {
                    zenText.classList.add('active');
                } else {
                    zenText2.classList.add('active');
                }
            } else {
                zenText.classList.remove('active');
                zenText2.classList.remove('active');
            }

            dragonLerp += (isDragonMode ? 0.03 : -0.03);
            dragonLerp = Math.max(0, Math.min(1, dragonLerp));

            if (isDragonMode || dragonLerp > 0) {
                dragonSpine.unshift({x: mouse.x, y: mouse.y});
                if (dragonSpine.length > 45) dragonSpine.pop(); 
            }

            // 更新 3D 模式插值
            let target3D = is3DMode ? 1 : 0;
            lerp3D += (target3D - lerp3D) * 0.03;
            
            // 3D 星球自动缓动旋转
            if (is3DMode && !isDragging) {
                targetRotY += 0.003; 
            }
            rotX += (targetRotX - rotX) * 0.05;
            rotY += (targetRotY - rotY) * 0.05;

            // 摄像机景深平滑推进(产生平滑的缩放感)
            currentCameraZ += (targetCameraZ - currentCameraZ) * 0.08;

            angle += 0.0035; 
            const cx = width / 2;
            const cy = height / 2;

            // 第一阶段：更新所有粒子的物理位置和 3D 投影坐标
            for (let i = 0; i < particles.length; i++) {
                particles[i].update(cx, cy, angle);
            }

            // 第二阶段：如果处于 3D 模式，根据 Z 轴深度对粒子进行排序（近大远小，防止错误的遮挡）
            if (lerp3D > 0.01) {
                particles.sort((a, b) => b.projZ - a.projZ);
            }

            // 第三阶段：开启混合模式并绘制
            // 开启灵光模式时，改变 Canvas 的混合模式为“滤色(更亮)”
            ctx.globalCompositeOperation = isGlowMode ? 'lighter' : 'source-over';

            for (let i = 0; i < particles.length; i++) {
                particles[i].draw();
            }

            // 绘制完毕后恢复默认混合模式，以免影响背景绘制
            ctx.globalCompositeOperation = 'source-over';

            requestAnimationFrame(animate);
        }

        // 交互监听
        window.addEventListener('mousemove', (e) => {
            let dx = 0, dy = 0;
            if(lastMouse.x !== -1000) {
                dx = e.clientX - lastMouse.x;
                dy = e.clientY - lastMouse.y;
                let speed = Math.hypot(dx, dy);
                
                mouseEnergy += speed * 0.15;
                if (mouseEnergy > 250) mouseEnergy = 250; 

                if (is3DMode && isDragging && !ignoreDrag) {
                    // 3D 模式下拖动鼠标：全向旋转水墨星球
                    targetRotY += dx * 0.01;
                    targetRotX -= dy * 0.01;
                } else if (isDragging && !ignoreDrag) {
                    // 2D 模式下拖动鼠标：搅动水墨
                    mixEnergy += speed * 0.5; 
                    if (mixEnergy > 500) mixEnergy = 500;
                }
            }
            lastMouse.x = e.clientX;
            lastMouse.y = e.clientY;

            mouse.x = e.clientX;
            mouse.y = e.clientY;
            mouse.isActive = true;
        });

        // 模式切换按钮事件
        modeBtn.addEventListener('click', (e) => {
            currentMode = currentMode === 1 ? 2 : 1;
            modeBtn.innerText = currentMode === 1 ? '当前：太极图' : '当前：阴阳流';
            
            mixEnergy = 0;
            isZenMode = false;
            ignoreDrag = false;
        });

        // 自定义颜色面板交互逻辑
        const colorToggle = document.getElementById('color-toggle');
        const colorPanel = document.getElementById('color-panel');
        
        colorToggle.addEventListener('click', () => {
            colorPanel.classList.toggle('show');
        });

        document.getElementById('bg-color-picker').addEventListener('input', (e) => {
            let rgb = hexToRgb(e.target.value);
            if(rgb) { bgBaseR = rgb.r; bgBaseG = rgb.g; bgBaseB = rgb.b; }
        });

        document.getElementById('color1-picker').addEventListener('input', (e) => {
            let rgb = hexToRgb(e.target.value);
            if(rgb) {
                c1BaseR = rgb.r; c1BaseG = rgb.g; c1BaseB = rgb.b;
                for(let p of particles) {
                    if (p.isLeft) {
                        p.base_r = Math.max(0, Math.min(255, c1BaseR + (Math.random() * 30 - 15)));
                        p.base_g = Math.max(0, Math.min(255, c1BaseG + (Math.random() * 30 - 15)));
                        p.base_b = Math.max(0, Math.min(255, c1BaseB + (Math.random() * 30 - 15)));
                    }
                }
            }
        });

        document.getElementById('color2-picker').addEventListener('input', (e) => {
            let rgb = hexToRgb(e.target.value);
            if(rgb) {
                c2BaseR = rgb.r; c2BaseG = rgb.g; c2BaseB = rgb.b;
                for(let p of particles) {
                    if (!p.isLeft) {
                        p.base_r = Math.max(0, Math.min(255, c2BaseR + (Math.random() * 20 - 10)));
                        p.base_g = Math.max(0, Math.min(255, c2BaseG + (Math.random() * 20 - 10)));
                        p.base_b = Math.max(0, Math.min(255, c2BaseB + (Math.random() * 20 - 10)));
                    }
                }
            }
        });

        // 灵光特效按钮逻辑
        const glowToggleBtn = document.getElementById('glow-toggle');
        glowToggleBtn.addEventListener('click', () => {
            isGlowMode = !isGlowMode;
            glowToggleBtn.classList.toggle('active');
        });

        // 3D 模式按钮逻辑
        const mode3dToggleBtn = document.getElementById('mode-3d-toggle');
        const hint3DText = document.getElementById('hint-3d');

        mode3dToggleBtn.addEventListener('click', () => {
            is3DMode = !is3DMode;
            mode3dToggleBtn.classList.toggle('active');
            
            // 开启 3D 时，赋予一个初始的微小倾斜角，以便立刻看出立体效果
            if (is3DMode) {
                targetRotX = -0.4;
                targetRotY = 0.5;
                // 显示操作提示
                hint3DText.style.opacity = '1';
                setTimeout(() => hint3DText.style.opacity = '0', 5000);
            } else {
                targetRotX = 0;
                targetRotY = 0;
                targetCameraZ = 200; // 退出3D时恢复默认摄像机距离
                hint3DText.style.opacity = '0';
            }
        });

        // 鼠标滚轮无级缩放 3D 空间
        window.addEventListener('wheel', (e) => {
            if (is3DMode) {
                // 滚轮向下为正(拉远)，向上为负(拉近)
                targetCameraZ += e.deltaY * 0.8;
                // 限制摄像机推拉极限：-250 (深深冲入星球内部) ~ 2500 (极远观测)
                targetCameraZ = Math.max(-250, Math.min(2500, targetCameraZ));
            }
        }, {passive: true});

        // 作者弹窗交互逻辑
        const authorBtn = document.getElementById('author-btn');
        const authorModal = document.getElementById('author-modal');
        const closeAuthor = document.getElementById('close-author');

        authorBtn.addEventListener('click', () => {
            authorModal.classList.add('show');
        });

        closeAuthor.addEventListener('click', () => {
            authorModal.classList.remove('show');
        });

        // 点击遮罩层关闭弹窗
        authorModal.addEventListener('click', (e) => {
            if (e.target === authorModal) {
                authorModal.classList.remove('show');
            }
        });

        window.addEventListener('mousedown', (e) => {
            // 忽略点击各种按钮、面板和弹窗时的太极反转事件，防止冲突
            if (e.target.closest('#mode-btn') || 
                e.target.closest('#color-controls') || 
                e.target.closest('#glow-controls') || 
                e.target.closest('#d3-controls') ||
                e.target.closest('#author-btn') ||
                e.target.closest('#author-modal')) return;

            isDragging = true;
            
            // 不同模式执行不同的点击反转逻辑
            if (currentMode === 1) {
                targetAngleOffset += Math.PI; // 太极阴阳旋转
            } else {
                targetColorSwap = targetColorSwap === 0 ? 1 : 0; // 流水模式黑白平滑互换
            }

            if (is3DMode) {
                // 3D模式下：取消凌乱的粒子炸开，改为优雅的“空间脉冲”视效
                currentCameraZ += 150; // 瞬间拉远镜头，配合原有的平滑推进，形成一种“空间心跳”的冲击反馈
                rotY -= 0.15;          // 给星球一个轻微的受击偏转力
                rotX -= 0.05;
            } else {
                // 2D模式下：保留原有的局部粒子炸开效果
                for(let p of particles) {
                    let dx = p.x - e.clientX;
                    let dy = p.y - e.clientY;
                    let dist = Math.hypot(dx, dy);
                    if(dist < 200) {
                        let force = (200 - dist) / 8;
                        p.vx += (dx / dist) * force;
                        p.vy += (dy / dist) * force;
                    }
                }
            }
        });

        window.addEventListener('mouseup', () => {
            isDragging = false;
            ignoreDrag = false; 
        });

        // 移动端支持长按搅动与点击
        window.addEventListener('touchstart', (e) => {
            // 忽略触摸各种按钮、面板和弹窗时的事件
            if (e.target.closest('#mode-btn') || 
                e.target.closest('#color-controls') || 
                e.target.closest('#glow-controls') || 
                e.target.closest('#d3-controls') ||
                e.target.closest('#author-btn') ||
                e.target.closest('#author-modal')) return;

            // 拦截并处理双指捏合缩放
            if (e.touches.length === 2 && is3DMode) {
                initialPinchDist = Math.hypot(
                    e.touches[0].clientX - e.touches[1].clientX,
                    e.touches[0].clientY - e.touches[1].clientY
                );
                return; 
            }

            // 单指触发旋转或排斥
            if (e.touches.length === 1) {
                isDragging = true;
                mouse.x = lastMouse.x = e.touches[0].clientX;
                mouse.y = lastMouse.y = e.touches[0].clientY;
                mouse.isActive = true;
                
                if (currentMode === 1) {
                    targetAngleOffset += Math.PI;
                } else {
                    targetColorSwap = targetColorSwap === 0 ? 1 : 0;
                }

                if (is3DMode) {
                    // 3D模式下的空间脉冲反馈
                    currentCameraZ += 120;
                    rotY -= 0.15;
                    rotX -= 0.05;
                } else {
                    // 2D模式炸开效果 (为移动端补齐一致的体验)
                    for(let p of particles) {
                        let dx = p.x - mouse.x;
                        let dy = p.y - mouse.y;
                        let dist = Math.hypot(dx, dy);
                        if(dist < 200) {
                            let force = (200 - dist) / 8;
                            p.vx += (dx / dist) * force;
                            p.vy += (dy / dist) * force;
                        }
                    }
                }
            }
        }, {passive: true});

        window.addEventListener('touchmove', (e) => {
            // 处理移动端的双指捏合缩放
            if (e.touches.length === 2 && is3DMode) {
                let currentDist = Math.hypot(
                    e.touches[0].clientX - e.touches[1].clientX,
                    e.touches[0].clientY - e.touches[1].clientY
                );
                let diff = initialPinchDist - currentDist; // diff>0为捏合拉远，diff<0为张开拉近
                targetCameraZ += diff * 2.5;
                targetCameraZ = Math.max(-250, Math.min(2500, targetCameraZ));
                initialPinchDist = currentDist;
                return;
            }

            // 单指滑动旋转逻辑
            if (e.touches.length === 1 && isDragging && !ignoreDrag) {
                let currX = e.touches[0].clientX;
                let currY = e.touches[0].clientY;
                
                let dx = currX - lastMouse.x;
                let dy = currY - lastMouse.y;
                let speed = Math.hypot(dx, dy);
                
                if (is3DMode) {
                    targetRotY += dx * 0.01;
                    targetRotX -= dy * 0.01;
                } else {
                    mixEnergy += speed * 0.5;
                    if (mixEnergy > 500) mixEnergy = 500;
                }
                
                lastMouse.x = currX;
                lastMouse.y = currY;
                mouse.x = currX;
                mouse.y = currY;
            }
        }, {passive: true});

        window.addEventListener('touchend', () => {
            isDragging = false;
            mouse.isActive = false;
            ignoreDrag = false; 
        });

        window.addEventListener('resize', () => {
            resize();
        });

        resize();
        initParticles();
        animate();

    </script>
